# Шпаргалка по Git для работы над проектом

```bash
#- Проверить на какой ты ветке
git branch

#- Создать свою ветку для задачи и сразу переключиться на неё
git checkout -b feature/название-фичи

#- Удалить ветку
git branch -d  ветка

#- Переключиться на основную ветку (main или master)
git checkout main

#- Забрать свежие изменения с удалённого репозитория
git pull origin main

#- Добавить изменения
git add .

#- Сделать коммит изменения с понятным сообщением
git commit -m "Доработал файл для функций о создании функций"

git commit -m "
- Добавленно переключение между страницами
- Кнопки вынесены в js скрипт теперь они на каждой странице
- добавленно поле разработки
"

#- Отправить свою ветку на GitHub (первый раз)
git push origin ветка

#- Слияние
git merge ветка
